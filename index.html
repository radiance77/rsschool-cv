    function indexEqualsValue(arr) {

            middle = Math.floor((start + end) / 2)
            if (arr[middle] === middle) {
                array.push(middle);
                end = middle;
            } else if (arr[middle] > middle) {
                end = middle;
            } else if (arr[middle] < middle) {
                start = middle;
            }
        }
                            
        if (arr[0] === 0) {
            return 0;
        }
                          
        if(array.length > 0) {
            return Math.min(...array);
        }else{
            return -1;
        }
